設定 Supabase 後端

請按照以下步驟設定：

  1. 前往 Supabase.com 註冊免費帳號
  2. 建立新專案（New Project）
  3. 在 SQL Editor 中執行以下指令： 
    -- 建立用戶表
CREATE TABLE IF NOT EXISTS users (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  username TEXT UNIQUE NOT NULL,
  password TEXT NOT NULL,
  is_admin BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 建立作業資料表
CREATE TABLE IF NOT EXISTS homework_data (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  data JSONB NOT NULL,
  updated_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(user_id)
);

-- 建立索引
CREATE INDEX IF NOT EXISTS idx_homework_user ON homework_data(user_id);
CREATE INDEX IF NOT EXISTS idx_users_username ON users(username);

-- 啟用 RLS (如果還沒啟用)
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
ALTER TABLE homework_data ENABLE ROW LEVEL SECURITY;

-- 刪除舊政策（如果存在）
DROP POLICY IF EXISTS "Users can read own data" ON users;
DROP POLICY IF EXISTS "Users can insert own data" ON users;
DROP POLICY IF EXISTS "Users can update own data" ON users;
DROP POLICY IF EXISTS "Homework data access" ON homework_data;

-- 建立新的 RLS 政策
CREATE POLICY "Users can read own data" ON users
  FOR SELECT USING (true);

CREATE POLICY "Users can insert own data" ON users
  FOR INSERT WITH CHECK (true);

CREATE POLICY "Users can update own data" ON users
  FOR UPDATE USING (true);

CREATE POLICY "Homework data access" ON homework_data
  FOR ALL USING (true);
  4. 在 Settings → API 中找到：
    • Project URL（格式：https://xxxxx.supabase.co）
    • anon public key（很長的一串文字）
